412 bool cltool_replayDataLog()
413 {
414  if (g_commandLineOptions.logPath.length() == 0)
415  {
416  cout << "Please specify the replay log path!" << endl;
417  return false;
418  }
419 
420  cISLogger logger;
421  if (!logger.LoadFromDirectory(g_commandLineOptions.logPath, cISLogger::ParseLogType(g_commandLineOptions.logType), { "ALL" }))
422  {
423  cout << "Failed to load log files: " << g_commandLineOptions.logPath << endl;
424  return false;
425  }
426 
427  cout << "Replaying log files: " << g_commandLineOptions.logPath << endl;
428  p_data_t *data;
429  while ((data = logger.ReadData()) != NULL)
430  {
431  g_inertialSenseDisplay.ProcessData(data, g_commandLineOptions.replayDataLog, g_commandLineOptions.replaySpeed);
432 
433 // if (data->hdr.id == DID_GPS1_RAW)
434 // {
435 // // Insert your code for processing data here
436 // }
437  }
438 
439  cout << "Done replaying log files: " << g_commandLineOptions.logPath << endl;
440  g_inertialSenseDisplay.Goodbye();
441  return true;
442 }
